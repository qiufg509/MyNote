```bat
@echo off
goto start
:sub
set /p keycode="KeyCode(q退出):"
if "%keycode%"=="q" goto end
if %keycode% gtr 0  (
	if %keycode% lss 260 (
		adb shell input keyevent %keycode%
		goto :eof
	)
)
echo invalid keycode.
goto :eof

:start
for /l %%i in (1,1,5) do (
	call :sub
)
goto start

:end
exit
```
